  <script>

    // Read URL params

    const params = new URLSearchParams(window.location.search);


    // Replace labels with URL values

    document.getElementById('q1Label').innerText = params.get('q1') || 'Static Question 1';

    document.getElementById('q2Label').innerText = params.get('q2') || 'Static Question 2';

    document.getElementById('q3Label').innerText = params.get('q3') || 'Static Question 3';

    document.getElementById('dq1Label').innerText = params.get('dq1') || 'Dynamic Question 1';

    document.getElementById('dq2Label').innerText = params.get('dq2') || 'Dynamic Question 2';


    // Submit form

    document.getElementById('insightForm').addEventListener('submit', async (e) => {

      e.preventDefault();


      const formData = new FormData(e.target);

      const json = Object.fromEntries(formData.entries());


      // Send to webhook (replace with your n8n webhook URL)

      const webhookUrl = 'https://YOUR-N8N-URL/webhook/insight-response';


      const res = await fetch(webhookUrl, {

        method: 'POST',

        headers: { 'Content-Type': 'application/json' },

        body: JSON.stringify(json)

      });


      document.getElementById('status').innerText = res.ok

        ? 'Thanks! Your response has been recorded.'

        : 'Oops, something went wrong.';

    });

  </script>
